* bind()
* children()
* clone()
* contents()
* css()
* data()

* empty()
* eq()
* find()
* hasClass()
* html()
* next()
* on()
* off()
* one()
* parent()
* prepend()
* prop()
* ready()
* remove()
* removeAttr()
* removeClass()
* removeData()
* replaceWith()
* text()
* toggleClass()
